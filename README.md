
```sh
  TINDER_HOST = "https://api.gotinder.com"
  TINDER_UID = "<your-tinder-uid>"
  FACEBOOK_UID = "<your-facebook-uid>"
  FACEBOOK_COOKIES = '[{"domain":"www.facebook.com","name":"<cookie-name>","value":"<cookie-value>"}]'
```
